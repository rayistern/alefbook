%%% ---- Custom Seder-step command (kids: stars instead of ornaments) ----
%%% \sedersection{Hebrew Name}{Transliteration}{English Subtitle}{step number}
\newif\ifsectionpage
\newcommand{\sedersection}[4]{%
  \needspace{3.5in}%
  \sectionpagetrue
  \markboth{#2\enspace---\enspace#3}{}%
  \par\vspace{12pt}
  %% decorative header (stars + dashed rule)
  \begin{center}
    \begin{tikzpicture}
      \node at (-3.0,0) {\textcolor{sedergold}{\large$\bigstar$}};
      \draw[sedergold, line width=0.6pt, densely dashed] (-2.4,0) -- (-0.15,0);
      \node[inner sep=0pt] at (0,0) {\textcolor{sedergold}{$\star$}};
      \draw[sedergold, line width=0.6pt, densely dashed] (0.15,0) -- (2.4,0);
      \node at (3.0,0) {\textcolor{sedergold}{\large$\bigstar$}};
    \end{tikzpicture}
  \end{center}
  \vspace{2pt}
  %% Titles (kids fonts)
  \begin{center}
    {\hebrewfontsanslarge\textcolor{sederblue}{\beginR #1\endR}}\\[4pt]
    {\kidssteptitlefont\Large\textcolor{sederblue}{#2}}\\[2pt]
    {\kidstitlefont\normalsize\textcolor{sederwine}{#3}}
  \end{center}
  \vspace{1pt}
  %% decorative bottom rule (dotted)
  \begin{center}
    \begin{tikzpicture}
      \draw[sedergold, line width=0.8pt, dotted] (-2.5,0) -- (2.5,0);
    \end{tikzpicture}
  \end{center}
  \vspace{4pt}
}
